1//! Pkarr based node discovery for iroh, supporting both relay servers and the DHT.
2//!
3//! This module contains pkarr-based node discovery for iroh which can use both pkarr
4//! relay servers as well as the Mainline DHT directly.  See the [pkarr module] for an
5//! overview of pkarr.
6//!
7//! [pkarr module]: super

29/// Republish delay for the DHT.
30///
31/// This is only for when the info does not change.  If the info changes, it will be
32/// published immediately.
33const REPUBLISH_DELAY: Duration = Duration::from_secs(60 * 60);
34/// Initial publish delay.
35///
36/// This is to avoid spamming the DHT when there are frequent network changes at startup.
37const INITIAL_PUBLISH_DELAY: Duration = Duration::from_millis(500);
38
39/// Pkarr Mainline DHT and relay server node discovery.
40///
41/// It stores node addresses in DNS records, signed by the node's private key, and publishes
42/// them to the BitTorrent Mainline DHT.  See the [pkarr module] for more details.
43///
44/// This implements the [`Discovery`] trait to be used as a node discovery service which can
45/// be used as both a publisher and resolver.  Calling [`DhtDiscovery::publish`] will start
46/// a background task that periodically publishes the node address.
47///
48/// [pkarr module]: super
49#[derive(Debug, Clone)]
50pub struct DhtDiscovery(Arc<Inner>);
51
52impl Default for DhtDiscovery {
53    fn default() -> Self {
54        Self::builder().build().expect("valid builder")
55    }
56}

58#[derive(derive_more::Debug)]
59struct Inner {
60    /// Pkarr client for interacting with the DHT.
61    pkarr: PkarrClient,
62    /// The background task that periodically publishes the node address.
63    ///
64    /// Due to [`AbortOnDropHandle`], this will be aborted when the discovery is dropped.
65    task: Mutex<Option<AbortOnDropHandle<()>>>,
66    /// Optional keypair for signing the DNS packets.
67    ///
68    /// If this is None, the node will not publish its address to the DHT.
69    secret_key: Option<SecretKey>,
70    /// Optional pkarr relay URL to use.
71    relay_url: Option<Url>,
72    /// Time-to-live value for the DNS packets.
73    ttl: u32,
74    /// True to include the direct addresses in the DNS packet.
75    include_direct_addresses: bool,
76    /// Initial delay before the first publish.
77    initial_publish_delay: Duration,
78    /// Republish delay for the DHT.
79    republish_delay: Duration,
80}
81
82impl Inner {
83    async fn resolve_pkarr(
84        &self,
85        key: pkarr::PublicKey,
86    ) -> Option<Result<DiscoveryItem, DiscoveryError>> {
87        tracing::info!(
88            "resolving {} from relay and DHT {:?}",
89            key.to_z32(),
90            self.relay_url
91        );
92
93        let maybe_packet = self.pkarr.resolve(&key).await;
94        match maybe_packet {
95            Some(signed_packet) => match NodeInfo::from_pkarr_signed_packet(&signed_packet) {
96                Ok(node_info) => {
97                    tracing::info!("discovered node info {:?}", node_info);
98                    Some(Ok(DiscoveryItem::new(node_info, "pkarr", None)))
99                }
100                Err(_err) => {
101                    tracing::debug!("failed to parse signed packet as node info");
102                    None
103                }
104            },
105            None => {
106                tracing::debug!("no signed packet found");
107                None
108            }
109        }
110    }
111}

113/// Builder for [`DhtDiscovery`].
114///
115/// By default, publishing to the DHT is enabled, and relay publishing is disabled.
116#[derive(Debug)]
117pub struct Builder {
118    client: Option<PkarrClient>,
119    secret_key: Option<SecretKey>,
120    ttl: Option<u32>,
121    pkarr_relay: Option<Url>,
122    dht: bool,
123    include_direct_addresses: bool,
124    initial_publish_delay: Duration,
125    republish_delay: Duration,
126    enable_publish: bool,
127}
128
129impl Default for Builder {
130    fn default() -> Self {
131        Self {
132            client: None,
133            secret_key: None,
134            ttl: None,
135            pkarr_relay: None,
136            dht: true,
137            include_direct_addresses: false,
138            initial_publish_delay: INITIAL_PUBLISH_DELAY,
139            republish_delay: REPUBLISH_DELAY,
140            enable_publish: true,

210    pub fn no_publish(mut self) -> Self {
211        self.enable_publish = false;
212        self

215    /// Builds the discovery mechanism.
216    pub fn build(self) -> Result<DhtDiscovery, IntoDiscoveryError> {
217        if !(self.dht || self.pkarr_relay.is_some()) {
218            return Err(IntoDiscoveryError::from_err(
219                "pkarr",
220                std::io::Error::other("at least one of DHT or relay must be enabled"),
221            ));
222        }
223        let pkarr = match self.client {
224            Some(client) => client,
225            None => {
226                let mut builder = PkarrClient::builder();
227                builder.no_default_network();
228                if self.dht {
229                    builder.dht(|x| x);
230                }
231                if let Some(url) = &self.pkarr_relay {
232                    builder
233                        .relays(&[url.clone()])
234                        .map_err(|e| IntoDiscoveryError::from_err("pkarr", e))?;
235                }
236                builder
237                    .build()
238                    .map_err(|e| IntoDiscoveryError::from_err("pkarr", e))?
239            }
240        };
241        let ttl = self.ttl.unwrap_or(DEFAULT_PKARR_TTL);
242        let include_direct_addresses = self.include_direct_addresses;
243        let secret_key = self.secret_key.filter(|_| self.enable_publish);
244
245        Ok(DhtDiscovery(Arc::new(Inner {
246            pkarr,
247            ttl,
248            relay_url: self.pkarr_relay,
249            include_direct_addresses,
250            secret_key,
251            initial_publish_delay: self.initial_publish_delay,
252            republish_delay: self.republish_delay,
253            task: Default::default(),
254        })))
255    }
256}

267impl DhtDiscovery {
268    /// Creates a new builder for [`DhtDiscovery`].
269    pub fn builder() -> Builder {
270        Builder::default()
271    }
272
273    /// Periodically publishes the node address to the DHT and/or relay.
274    async fn publish_loop(self, keypair: SecretKey, signed_packet: SignedPacket) {
275        let this = self;
276        let z32 = pkarr::PublicKey::try_from(keypair.public().as_bytes())
277            .expect("valid public key")
278            .to_z32();
279        // initial delay. If the task gets aborted before this delay is over,
280        // we have not published anything to the DHT yet.
281        time::sleep(this.0.initial_publish_delay).await;
282        loop {
283            // TODO: publish takes a compare-and-swap timestamp, make use of that.
284            let res = this.0.pkarr.publish(&signed_packet, None).await;
285            match res {
286                Ok(()) => {
287                    tracing::debug!("pkarr publish success. published under {z32}",);
288                }
289                Err(e) => {
290                    // we could do a smaller delay here, but in general DHT publish
291                    // not working is due to a network issue, and if the network changes
292                    // the task will be restarted anyway.
293                    //
294                    // Being unable to publish to the DHT is something that is expected
295                    // to happen from time to time, so this does not warrant a error log.
296                    tracing::warn!("pkarr publish error: {}", e);
297                }
298            }
299            time::sleep(this.0.republish_delay).await;
300        }
301    }
302}
303
304impl Discovery for DhtDiscovery {
305    fn publish(&self, data: &NodeData) {
306        let Some(keypair) = &self.0.secret_key else {
307            tracing::debug!("no keypair set, not publishing");
308            return;
309        };
310        tracing::debug!("publishing {data:?}");
311        let mut info = NodeInfo::from_parts(keypair.public(), data.clone());
312        if !self.0.include_direct_addresses {
313            info.clear_direct_addresses();
314        }
315        let Ok(signed_packet) = info.to_pkarr_signed_packet(keypair, self.0.ttl) else {
316            tracing::warn!("failed to create signed packet");
317            return;
318        };
319        let this = self.clone();
320        let curr = task::spawn(this.publish_loop(keypair.clone(), signed_packet));
321        let mut task = self.0.task.lock().expect("poisoned");
322        *task = Some(AbortOnDropHandle::new(curr));
323    }
324
325    fn resolve(&self, node_id: NodeId) -> Option<BoxStream<Result<DiscoveryItem, DiscoveryError>>> {
326        let pkarr_public_key =
327            pkarr::PublicKey::try_from(node_id.as_bytes()).expect("valid public key");
328        tracing::info!("resolving {} as {}", node_id, pkarr_public_key.to_z32());
329        let discovery = self.0.clone();
330        let stream = n0_future::stream::once_future(async move {
331            discovery.resolve_pkarr(pkarr_public_key).await
332        })
333        .filter_map(|x| x)
334        .boxed();
335        Some(stream)
336    }
337}

340mod tests {
341    use std::collections::BTreeSet;
342
343    use iroh_base::RelayUrl;
344    use n0_snafu::{Result, ResultExt};
345    use tracing_test::traced_test;
346
347    use super::*;
348    use crate::Endpoint;
349
350    #[tokio::test]
351    #[ignore = "flaky"]
352    #[traced_test]
353    async fn dht_discovery_smoke() -> Result {
354        let ep = Endpoint::builder().bind().await?;
355        let secret = ep.secret_key().clone();
356        let testnet = pkarr::mainline::Testnet::new_async(3).await.e()?;
357        let client = pkarr::Client::builder()
358            .dht(|builder| builder.bootstrap(&testnet.bootstrap))
359            .build()
360            .e()?;
361        let discovery = DhtDiscovery::builder()
362            .secret_key(secret.clone())
363            .initial_publish_delay(Duration::ZERO)
364            .client(client)
365            .build()?;
366
367        let relay_url: RelayUrl = Url::parse("https://example.com").e()?.into();
368
369        let data = NodeData::default().with_relay_url(Some(relay_url.clone()));
370        discovery.publish(&data);
371
372        // publish is fire and forget, so we have no way to wait until it is done.
373        tokio::time::timeout(Duration::from_secs(30), async move {
374            loop {
375                tokio::time::sleep(Duration::from_millis(200)).await;
376                let mut found_relay_urls = BTreeSet::new();
377                let items = discovery
378                    .resolve(secret.public())
379                    .unwrap()
380                    .collect::<Vec<_>>()
381                    .await;
382                for item in items.into_iter().flatten() {
383                    if let Some(url) = item.relay_url() {
384                        found_relay_urls.insert(url.clone());
385                    }
386                }
387                if found_relay_urls.contains(&relay_url) {
388                    break;
389                }
390            }
391        })
392        .await
393        .expect("timeout, relay_url not found on DHT");
394        Ok(())
395    }
