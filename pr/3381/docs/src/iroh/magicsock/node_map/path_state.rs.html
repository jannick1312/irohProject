17/// State about a particular path to another [`NodeState`].
18///
19/// This state is used for both the relay path and any direct UDP paths.
20///
21/// [`NodeState`]: super::node_state::NodeState
22#[derive(Debug, Clone, PartialEq, Eq)]
23pub(super) struct PathState {
24    /// The node for which this path exists.
25    node_id: NodeId,
26    /// The path this applies for.
27    path: SendAddr,
28
29    /// The time this endpoint was last advertised via a call-me-maybe DISCO message.
30    pub(super) call_me_maybe_time: Option<Instant>,
31
32    /// When the last payload data was **received** via this path.
33    ///
34    /// This excludes DISCO messages.
35    pub(super) last_payload_msg: Option<Instant>,
36    /// Sources is a map of [`Source`]s to [`Instant`]s, keeping track of all the ways we have
37    /// learned about this path
38    ///
39    /// We keep track of only the latest [`Instant`] for each [`Source`], keeping the size of
40    /// the map of sources down to one entry per type of source.
41    pub(super) sources: HashMap<Source, Instant>,
42}
43
44impl PathState {
45    pub(super) fn new(node_id: NodeId, path: SendAddr, source: Source, now: Instant) -> Self {
46        let mut sources = HashMap::new();
47        sources.insert(source, now);
48        Self {
49            node_id,
50            path,
51            call_me_maybe_time: None,
52            last_payload_msg: None,
53            sources,
54        }
55    }
56
57    pub(super) fn udp_addr(&self) -> Option<SocketAddr> {
58        match self.path {
59            SendAddr::Udp(addr) => Some(addr),
60            SendAddr::Relay(_) => None,
61        }
62    }
63
64    pub(super) fn with_last_payload(
65        node_id: NodeId,
66        path: SendAddr,
67        source: Source,
68        now: Instant,
69    ) -> Self {
70        let mut sources = HashMap::new();
71        sources.insert(source, now);
72        PathState {
73            node_id,
74            path,
75            call_me_maybe_time: None,
76            last_payload_msg: Some(now),
77            sources,
78        }
79    }
80
81    /// Check whether this path is considered active.
82    ///
83    /// Active means the path has received payload messages within the last
84    /// [`SESSION_ACTIVE_TIMEOUT`].
85    ///
86    /// Note that a path might be alive but not active if it's contactable but not in
87    /// use.
88    pub(super) fn is_active(&self) -> bool {
89        self.last_payload_msg
90            .as_ref()
91            .map(|instant| instant.elapsed() <= SESSION_ACTIVE_TIMEOUT)
92            .unwrap_or(false)
93    }
94
95    /// Reports the last instant this path was considered alive.
96    ///
97    /// Alive means the path is considered in use by the remote endpoint.  Either because we
98    /// received a payload message, a DISCO message (ping, pong) or it was advertised in a
99    /// call-me-maybe message.
100    ///
101    /// This is the most recent instant between:
102    /// - when last pong was received.
103    /// - when this path was last advertised in a received CallMeMaybe message.
104    /// - When the last payload transmission occurred.
105    /// - when the last ping from them was received.
106    pub(super) fn last_alive(&self) -> Option<Instant> {
107        self.last_payload_msg
108            .as_ref()
109            .into_iter()
110            .chain(self.call_me_maybe_time.as_ref())
111            .max()
112            .copied()
113    }
114
115    /// The last control or DISCO message **about** this path.
116    ///
117    /// This is the most recent instant among:
118    /// - when this path was last advertised in a received CallMeMaybe message.
119    /// - when the last ping from them was received.
120    ///
121    /// Returns the time elapsed since the last control message, and the type of control message.
122    pub(super) fn last_control_msg(&self, now: Instant) -> Option<(Duration, ControlMsg)> {
123        // get every control message and assign it its kind
124        let last_call_me_maybe = self
125            .call_me_maybe_time
126            .as_ref()
127            .map(|call_me| (*call_me, ControlMsg::CallMeMaybe));
128
129        last_call_me_maybe
130            .into_iter()
131            .max_by_key(|(instant, _kind)| *instant)
132            .map(|(instant, kind)| (now.duration_since(instant), kind))
133    }
134
135    pub(super) fn add_source(&mut self, source: Source, now: Instant) {
136        self.sources.insert(source, now);
137    }
138
139    pub(super) fn clear(&mut self) {
140        self.call_me_maybe_time = None;
141    }
142
143    fn summary(&self, mut w: impl std::fmt::Write) -> std::fmt::Result {
144        write!(w, "{{ ")?;
145        if self.is_active() {
146            write!(w, "active ")?;
147        }
148        if let Some(last_source) = self.sources.iter().max_by_key(|&(_, instant)| instant) {
149            write!(
150                w,
151                "last-source: {}({:?} ago)",
152                last_source.0,
153                last_source.1.elapsed()
154            )?;
155        }
156        write!(w, "}}")
157    }
158}

160// TODO: Make an `EndpointPaths` struct and do things nicely.
161pub(super) fn summarize_node_paths(paths: &BTreeMap<IpPort, PathState>) -> String {
162    use std::fmt::Write;
163
164    let mut w = String::new();
165    write!(&mut w, "[").ok();
166    for (i, (ipp, state)) in paths.iter().enumerate() {
167        if i > 0 {
168            write!(&mut w, ", ").ok();
169        }
170        write!(&mut w, "{ipp}").ok();
171        state.summary(&mut w).ok();
172    }
173    write!(&mut w, "]").ok();
174    w
175}
