13/// A net_report report.
14#[derive(Default, Debug, PartialEq, Eq, Clone)]
15pub struct Report {
16    /// A QAD IPv4 round trip completed.
17    pub udp_v4: bool,
18    /// A QAD IPv6 round trip completed.
19    pub udp_v6: bool,
20    /// Whether the reported public address differs when probing different servers (on IPv4).
21    pub mapping_varies_by_dest_ipv4: Option<bool>,
22    /// Whether the reported public address differs when probing different servers (on IPv6).
23    pub mapping_varies_by_dest_ipv6: Option<bool>,
24    /// Whether the reported public port differs when probing different destination ports (on IPv4).
25    pub mapping_varies_by_dest_port_ipv4: Option<bool>,
26    /// Whether the reported public port differs when probing different destination ports (on IPv6).
27    pub mapping_varies_by_dest_port_ipv6: Option<bool>,
28    /// Whether hairpinning (sending to own external address) works for IPv4.
29    pub hairpinning_ipv4: Option<bool>,
30    /// Whether hairpinning (sending to own external address) works for IPv6.
31    pub hairpinning_ipv6: Option<bool>,
32    /// Probe indicating the presence of port mapping protocols on the LAN.
33    /// `None` for unknown
34    pub preferred_relay: Option<RelayUrl>,
35    /// keyed by relay Url
36    pub relay_latency: RelayLatencies,
37    /// ip:port of global IPv4
38    pub global_v4: Option<SocketAddrV4>,
39    /// `[ip]:port` of global IPv6
40    pub global_v6: Option<SocketAddrV6>,
41    /// CaptivePortal is set when we think there's a captive portal that is
42    /// intercepting HTTP traffic.
43    pub captive_portal: Option<bool>,
44
45    // Internal tracking for port mapping variation detection
46    /// Tracks IPv4 public ports seen from different destination ports (dest_port -> public_port)
47    ipv4_port_mappings: BTreeMap<u16, u16>,
48    /// Tracks IPv6 public ports seen from different destination ports (dest_port -> public_port)
49    ipv6_port_mappings: BTreeMap<u16, u16>,
50}
51

58impl Report {
59    /// Do we have any indication that UDP is working?
60    pub fn has_udp(&self) -> bool {
61        self.udp_v4 || self.udp_v6
62    }
63
64    /// Whether the reported public address differs when probing different servers.
65    pub fn mapping_varies_by_dest(&self) -> Option<bool> {
66        match (
67            self.mapping_varies_by_dest_ipv4,
68            self.mapping_varies_by_dest_ipv6,
69        ) {
70            (Some(v4), Some(v6)) => Some(v4 || v6),
71            (None, Some(v6)) => Some(v6),
72            (Some(v4), None) => Some(v4),
73            (None, None) => None,
74        }
75    }
76
77    /// Whether the reported public port differs when probing different destination ports.
78    pub fn mapping_varies_by_dest_port(&self) -> Option<bool> {
79        match (
80            self.mapping_varies_by_dest_port_ipv4,
81            self.mapping_varies_by_dest_port_ipv6,
82        ) {
83            (Some(v4), Some(v6)) => Some(v4 || v6),
84            (None, Some(v6)) => Some(v6),
85            (Some(v4), None) => Some(v4),
86            (None, None) => None,
87        }
88    }
89
90    /// Whether hairpinning (sending to own external address) works.
91    pub fn hairpinning(&self) -> Option<bool> {
92        match (self.hairpinning_ipv4, self.hairpinning_ipv6) {
93            (Some(v4), Some(v6)) => Some(v4 || v6),
94            (None, Some(v6)) => Some(v6),
95            (Some(v4), None) => Some(v4),
96            (None, None) => None,
97        }
98    }

100    /// Updates a net_report [`Report`] with a new [`ProbeReport`].
101    pub(super) fn update(&mut self, report: &ProbeReport) {
102        match report {
103            ProbeReport::Https(report) => {
104                self.relay_latency
105                    .update_relay(report.node.clone(), report.latency, Probe::Https);
106            }
107            #[cfg(not(wasm_browser))]
108            ProbeReport::QadIpv4(report) => {
109                self.relay_latency.update_relay(
110                    report.node.clone(),
111                    report.latency,
112                    Probe::QadIpv4,
113                );
114                let SocketAddr::V4(ipp) = report.addr else {
115                    warn!("received IPv6 address from IPv4 QAD: {}", report.addr);
116                    return;
117                };
118
119                self.udp_v4 = true;
120
121                tracing::debug!(?self.global_v4, ?self.mapping_varies_by_dest_ipv4, %ipp,"got");
122                if let Some(global) = self.global_v4 {
123                    if global == ipp {
124                        if self.mapping_varies_by_dest_ipv4.is_none() {
125                            self.mapping_varies_by_dest_ipv4 = Some(false);
126                        }
127                    } else {
128                        self.mapping_varies_by_dest_ipv4 = Some(true);
129                        warn!("IPv4 address detected by QAD varies by destination");
130                    }
131                } else {
132                    self.global_v4 = Some(ipp);
133                }
134
135                let public_port = ipp.port();
136                let dest_port = report.dest_port;
137
138                if let Some(&existing_public_port) = self.ipv4_port_mappings.get(&dest_port) {
139                    if existing_public_port != public_port {
140                        self.mapping_varies_by_dest_port_ipv4 = Some(true);
141                        warn!(
142                            "IPv4 public port varies by destination port: dest {} -> public {} (was {})",
143                            dest_port, public_port, existing_public_port
144                        );
145                    } else if self.mapping_varies_by_dest_port_ipv4.is_none() {
146                        let has_different_mappings = self
147                            .ipv4_port_mappings
148                            .values()
149                            .any(|&port| port != public_port);
150                        self.mapping_varies_by_dest_port_ipv4 = Some(has_different_mappings);
151                    }
152                } else {
153                    self.ipv4_port_mappings.insert(dest_port, public_port);
154
155                    if self.ipv4_port_mappings.len() > 1 {
156                        let has_different_mappings = self
157                            .ipv4_port_mappings
158                            .values()
159                            .any(|&port| port != public_port);
160                        if has_different_mappings {
161                            self.mapping_varies_by_dest_port_ipv4 = Some(true);
162                        } else if self.mapping_varies_by_dest_port_ipv4.is_none() {
163                            self.mapping_varies_by_dest_port_ipv4 = Some(false);
164                        }
165                    }
166                }
167
168                if let Some(hairpinning_result) = report.hairpinning_works {
169                    self.hairpinning_ipv4 = Some(hairpinning_result);
170                }
171            }
172            #[cfg(not(wasm_browser))]
173            ProbeReport::QadIpv6(report) => {
174                self.relay_latency.update_relay(
175                    report.node.clone(),
176                    report.latency,
177                    Probe::QadIpv6,
178                );
179                let SocketAddr::V6(ipp) = report.addr else {
180                    warn!("received IPv4 address from IPv6 QAD: {}", report.addr);
181                    return;
182                };
183
184                self.udp_v6 = true;
185                tracing::debug!(?self.global_v6, ?self.mapping_varies_by_dest_ipv6, %ipp,"got");
186                if let Some(global) = self.global_v6 {
187                    if global == ipp {
188                        if self.mapping_varies_by_dest_ipv6.is_none() {
189                            self.mapping_varies_by_dest_ipv6 = Some(false);
190                        }
191                    } else {
192                        self.mapping_varies_by_dest_ipv6 = Some(true);
193                        warn!("IPv6 address detected by QAD varies by destination");
194                    }
195                } else {
196                    self.global_v6 = Some(ipp);
197                }
198
199                let public_port = ipp.port();
200                let dest_port = report.dest_port;
201
202                if let Some(&existing_public_port) = self.ipv6_port_mappings.get(&dest_port) {
203                    if existing_public_port != public_port {
204                        self.mapping_varies_by_dest_port_ipv6 = Some(true);
205                        warn!(
206                            "IPv6 public port varies by destination port: dest {} -> public {} (was {})",
207                            dest_port, public_port, existing_public_port
208                        );
209                    } else if self.mapping_varies_by_dest_port_ipv6.is_none() {
210                        let has_different_mappings = self
211                            .ipv6_port_mappings
212                            .values()
213                            .any(|&port| port != public_port);
214                        self.mapping_varies_by_dest_port_ipv6 = Some(has_different_mappings);
215                    }
216                } else {
217                    self.ipv6_port_mappings.insert(dest_port, public_port);
218
219                    if self.ipv6_port_mappings.len() > 1 {
220                        let has_different_mappings = self
221                            .ipv6_port_mappings
222                            .values()
223                            .any(|&port| port != public_port);
224                        if has_different_mappings {
225                            self.mapping_varies_by_dest_port_ipv6 = Some(true);
226                        } else if self.mapping_varies_by_dest_port_ipv6.is_none() {
227                            self.mapping_varies_by_dest_port_ipv6 = Some(false);
228                        }
229                    }
230                }
231
232                if let Some(hairpinning_result) = report.hairpinning_works {
233                    self.hairpinning_ipv6 = Some(hairpinning_result);
234                }
235            }
236        }
237    }

240/// Latencies per relay node.
241#[derive(Debug, Default, PartialEq, Eq, Clone)]
242pub struct RelayLatencies {
243    #[cfg(not(wasm_browser))]
244    ipv4: BTreeMap<RelayUrl, Duration>,
245    #[cfg(not(wasm_browser))]
246    ipv6: BTreeMap<RelayUrl, Duration>,
247    https: BTreeMap<RelayUrl, Duration>,
248}
249
250impl RelayLatencies {
251    /// Updates a relay's latency, if it is faster than before.
252    pub(super) fn update_relay(&mut self, url: RelayUrl, latency: Duration, probe: Probe) {
253        let list = match probe {
254            Probe::Https => &mut self.https,
255            #[cfg(not(wasm_browser))]
256            Probe::QadIpv4 | Probe::QadIpv4PortVariation => &mut self.ipv4,
257            #[cfg(not(wasm_browser))]
258            Probe::QadIpv6 | Probe::QadIpv6PortVariation => &mut self.ipv6,
259        };
260        let old_latency = list.entry(url).or_insert(latency);
261        if latency < *old_latency {
262            *old_latency = latency;
263        }
264    }
265
266    /// Merges another [`RelayLatencies`] into this one.
267    ///
268    /// For each relay the latency is updated using [`RelayLatencies::update_relay`].
269    pub(super) fn merge(&mut self, other: &RelayLatencies) {
270        for (url, latency) in other.https.iter() {
271            self.update_relay(url.clone(), *latency, Probe::Https);
272        }
273        #[cfg(not(wasm_browser))]
274        for (url, latency) in other.ipv4.iter() {
275            self.update_relay(url.clone(), *latency, Probe::QadIpv4);
276        }
277        #[cfg(not(wasm_browser))]
278        for (url, latency) in other.ipv6.iter() {
279            self.update_relay(url.clone(), *latency, Probe::QadIpv6);
280        }
281    }
282
283    /// Returns an iterator over all the relays and their latencies.
284    #[cfg(not(wasm_browser))]
285    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
286        self.https
287            .iter()
288            .chain(self.ipv4.iter())
289            .chain(self.ipv6.iter())
290            .map(|(k, v)| (k, *v))
291    }
292
293    /// Returns an iterator over all the relays and their latencies.
294    #[cfg(wasm_browser)]
295    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
296        self.https.iter().map(|(k, v)| (k, *v))
297    }
298
299    #[cfg(not(wasm_browser))]
300    pub(super) fn is_empty(&self) -> bool {
301        self.https.is_empty() && self.ipv4.is_empty() && self.ipv6.is_empty()
302    }
303
304    #[cfg(wasm_browser)]
305    pub(super) fn is_empty(&self) -> bool {
306        self.https.is_empty()
307    }
308
309    /// Returns the lowest latency across records.
310    pub(super) fn get(&self, url: &RelayUrl) -> Option<Duration> {
311        let mut list = Vec::with_capacity(3);
312        if let Some(val) = self.https.get(url) {
313            list.push(*val);
314        }
315        #[cfg(not(wasm_browser))]
316        if let Some(val) = self.ipv4.get(url) {
317            list.push(*val);
318        }
319        #[cfg(not(wasm_browser))]
320        if let Some(val) = self.ipv6.get(url) {
321            list.push(*val);
322        }
323        list.into_iter().min()
324    }

336    #[test]
337    fn test_port_mapping_variation_detection_ipv4() {
338        let mut report = Report::default();
339        let relay_url = RelayUrl::from(url::Url::parse("https://relay.example.com").unwrap());
340
341        // Test Case 1: Same public port for different destination ports -> false
342        let probe_port_7842 = ProbeReport::QadIpv4(QadProbeReport {
343            node: relay_url.clone(),
344            latency: Duration::from_millis(50),
345            addr: "203.0.113.42:54321".parse::<SocketAddr>().unwrap(),
346            dest_port: 7842,
347            hairpinning_works: None,
348        });
349
350        let probe_port_7843 = ProbeReport::QadIpv4(QadProbeReport {
351            node: relay_url.clone(),
352            latency: Duration::from_millis(52),
353            addr: "203.0.113.42:54321".parse::<SocketAddr>().unwrap(), // Same public port
354            dest_port: 7843,
355            hairpinning_works: None,
356        });
357
358        report.update(&probe_port_7842);
359        assert_eq!(
360            report.mapping_varies_by_dest_port_ipv4, None,
361            "Should be None with only one probe"
362        );
363
364        report.update(&probe_port_7843);
365        assert_eq!(
366            report.mapping_varies_by_dest_port_ipv4,
367            Some(false),
368            "Should be false when public ports are the same"
369        );
370
371        // Test Case 2: Different public port for third destination port -> true
372        let probe_port_7844 = ProbeReport::QadIpv4(QadProbeReport {
373            node: relay_url.clone(),
374            latency: Duration::from_millis(48),
375            addr: "203.0.113.42:54322".parse::<SocketAddr>().unwrap(), // Different public port
376            dest_port: 7844,
377            hairpinning_works: None,
378        });
379
380        report.update(&probe_port_7844);
381        assert_eq!(
382            report.mapping_varies_by_dest_port_ipv4,
383            Some(true),
384            "Should be true when public ports vary"
385        );
386
387        // Verify the mappings are tracked correctly
388        assert_eq!(report.ipv4_port_mappings.get(&7842), Some(&54321));
389        assert_eq!(report.ipv4_port_mappings.get(&7843), Some(&54321));
390        assert_eq!(report.ipv4_port_mappings.get(&7844), Some(&54322));
391    }
