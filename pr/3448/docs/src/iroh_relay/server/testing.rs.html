8/// Creates a [`rustls::ServerConfig`] and certificates suitable for testing.
9///
10/// - Uses a self signed certificate valid for the `"localhost"` and `"127.0.0.1"` domains.
11pub fn self_signed_tls_certs_and_config() -> (
12    Vec<rustls::pki_types::CertificateDer<'static>>,
13    rustls::ServerConfig,
14) {
15    let cert = rcgen::generate_simple_self_signed(vec![
16        "localhost".to_string(),
17        "127.0.0.1".to_string(),
18        "::1".to_string(),
19    ])
20    .expect("valid");
21    let rustls_cert = cert.cert.der();
22    let private_key = rustls::pki_types::PrivatePkcs8KeyDer::from(cert.signing_key.serialize_der());
23    let private_key = rustls::pki_types::PrivateKeyDer::from(private_key);
24    let certs = vec![rustls_cert.clone()];
25    let server_config = rustls::ServerConfig::builder_with_provider(std::sync::Arc::new(
26        rustls::crypto::ring::default_provider(),
27    ))
28    .with_safe_default_protocol_versions()
29    .expect("protocols supported by ring")
30    .with_no_client_auth();
31
32    let server_config = server_config
33        .with_single_cert(certs.clone(), private_key)
34        .expect("valid");
35    (certs, server_config)
36}

38/// Creates a [`TlsConfig`] suitable for testing.
39///
40/// - Uses a self signed certificate valid for the `"localhost"` and `"127.0.0.1"` domains.
41/// - Configures https to be served on an OS assigned port on ipv4.
42pub fn tls_config() -> TlsConfig<()> {
43    let (certs, server_config) = self_signed_tls_certs_and_config();
44    TlsConfig {
45        server_config,
46        cert: CertConfig::<(), ()>::Manual { certs },
47        https_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
48        quic_bind_addr: (Ipv4Addr::UNSPECIFIED, 0).into(),
49    }
50}
51
52/// Creates a [`RelayConfig`] suitable for testing.
53///
54/// - Binds http to an OS assigned port on ipv4.
55/// - Uses [`tls_config`] to enable TLS.
56/// - Uses default limits.
57pub fn relay_config() -> RelayConfig<()> {
58    RelayConfig {
59        http_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
60        tls: Some(tls_config()),
61        limits: Default::default(),
62        key_cache_capacity: Some(1024),
63        access: AccessConfig::Everyone,
64    }
65}

67/// Creates a [`QuicConfig`] suitable for testing.
68///
69/// - Binds to an OS assigned port on ipv4
70/// - Disables alternate port to avoid conflicts in parallel tests
71/// - Uses [`self_signed_tls_certs_and_config`] to create tls certificates
72pub fn quic_config() -> QuicConfig {
73    let (_, server_config) = self_signed_tls_certs_and_config();
74    QuicConfig {
75        bind_addr: (Ipv4Addr::UNSPECIFIED, 0).into(),
76        server_config,
77        alternate_port: AlternatePortConfig::Disabled, // Explicitly disable for tests
78    }
79}

81/// Creates a [`QuicConfig`] with port variation enabled for testing port detection logic.
82///
83/// - Same as regular quic_config but with alternate port enabled
84/// - Uses OS-assigned port (0) with alternate port enabled (main_port + 1)
85pub fn quic_config_with_port_variation() -> QuicConfig {
86    let (_, server_config) = self_signed_tls_certs_and_config();
87    QuicConfig {
88        bind_addr: (Ipv4Addr::UNSPECIFIED, 0).into(), // Let OS assign port
89        server_config,
90        alternate_port: AlternatePortConfig::Enabled, // Enable alternate port (main + 1)
91    }
92}

94/// Creates a [`ServerConfig`] suitable for testing.
95///
96/// - Relaying is enabled using [`relay_config`]
97/// - QUIC addr discovery is enabled with disabled alternate port to avoid conflicts.
98/// - Metrics are not enabled.
99pub fn server_config() -> ServerConfig<()> {
100    ServerConfig {
101        relay: Some(relay_config()),
102        quic: Some(quic_config()),
103        #[cfg(feature = "metrics")]
104        metrics_addr: None,
105    }
