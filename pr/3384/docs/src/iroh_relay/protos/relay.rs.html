1//! This module implements the relaying protocol used by the `server` and `client`.
2//!
3//! Protocol flow:
4//!
5//! Login:
6//!  * client connects
7//!  * -> client sends `FrameType::ClientInfo`
8//!
9//!  Steady state:
10//!  * server occasionally sends `FrameType::KeepAlive` (or `FrameType::Ping`)
11//!  * client responds to any `FrameType::Ping` with a `FrameType::Pong`
12//!  * clients sends `FrameType::SendPacket`
13//!  * server then sends `FrameType::RecvPacket` to recipient
14

54/// ProtocolVersion is bumped whenever there's a wire-incompatible change.
55///  - version 1 (zero on wire): consistent box headers, in use by employee dev nodes a bit
56///  - version 2: received packets have src addrs in FrameType::RecvPacket at beginning.
57///
58/// NOTE: we are technically running a modified version of the protocol.
59/// `FrameType::PeerPresent`, `FrameType::WatchConn`, `FrameType::ClosePeer`, have been removed.
60/// The server will error on that connection if a client sends one of these frames.
61/// We have split with the DERP protocol significantly starting with our relay protocol 3
62/// `FrameType::PeerPresent`, `FrameType::WatchConn`, `FrameType::ClosePeer`, `FrameType::ServerKey`, and `FrameType::ServerInfo` have been removed.
63/// The server will error on that connection if a client sends one of these frames.
64/// This materially affects the handshake protocol, and so relay nodes on version 3 will be unable to communicate
65/// with nodes running earlier protocol versions.
66pub(crate) const PROTOCOL_VERSION: usize = 3;
67
68/// Indicates this IS the client's home node
69const PREFERRED: u8 = 1u8;
70/// Indicates this IS NOT the client's home node
71const NOT_PREFERRED: u8 = 0u8;
72
73/// The one byte frame type at the beginning of the frame
74/// header. The second field is a big-endian u32 describing the
75/// length of the remaining frame (not including the initial 5 bytes)
76#[derive(Debug, PartialEq, Eq, num_enum::IntoPrimitive, num_enum::FromPrimitive, Clone, Copy)]
77#[repr(u8)]
78pub enum FrameType {
79    /// magic + 32b pub key + 24B nonce + bytes
80    ClientInfo = 2,
81    /// 32B dest pub key + packet bytes
82    SendPacket = 4,
83    /// v0/1 packet bytes, v2: 32B src pub key + packet bytes
84    RecvPacket = 5,
85    /// no payload, no-op (to be replaced with ping/pong)
86    KeepAlive = 6,
87    /// 1 byte payload: 0x01 or 0x00 for whether this is client's home node
88    NotePreferred = 7,
89    /// Sent from server to client to signal that a previous sender is no longer connected.
90    ///
91    /// That is, if A sent to B, and then if A disconnects, the server sends `FrameType::PeerGone`
92    /// to B so B can forget that a reverse path exists on that connection to get back to A
93    ///
94    /// 32B pub key of peer that's gone
95    PeerGone = 8,
96    /// Frames 9-11 concern meshing, which we have eliminated from our version of the protocol.
97    /// Messages with these frames will be ignored.
98    /// 8 byte ping payload, to be echoed back in FrameType::Pong
99    Ping = 12,
100    /// 8 byte payload, the contents of ping being replied to
101    Pong = 13,
102    /// Sent from server to client to tell the client if their connection is
103    /// unhealthy somehow.
104    ///
105    /// Currently this is used to indicate that the connection was closed because of authentication issues.
106    Health = 14,
107
108    /// Sent from server to client for the server to declare that it's restarting.
109    /// Payload is two big endian u32 durations in milliseconds: when to reconnect,
110    /// and how long to try total.
111    ///
112    /// Handled on the `[relay::Client]`, but currently never sent on the `[relay::Server]`
113    Restarting = 15,
114    /// Unknown frame type
115    #[num_enum(default)]
116    Unknown = 255,
117}

120    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
121        write!(f, "{self:?}")
122    }

181/// Writes complete frame, errors if it is unable to write within the given `timeout`.
182/// Ignores the timeout if `None`
183///
184/// Does not flush.
185#[cfg(feature = "server")]
186pub(crate) async fn write_frame<S: Sink<Frame, Error = std::io::Error> + Unpin>(
187    mut writer: S,
188    frame: Frame,
189    timeout: Option<Duration>,
190) -> Result<(), SendError> {
191    if let Some(duration) = timeout {
192        tokio::time::timeout(duration, writer.send(frame)).await??;
193    } else {
194        writer.send(frame).await?;
195    }
196
197    Ok(())
198}

200/// Writes a `FrameType::ClientInfo`, including the client's [`PublicKey`],
201/// and the client's [`ClientInfo`], sealed using the server's [`PublicKey`].
202///
203/// Flushes after writing.
204pub(crate) async fn send_client_key<S: Sink<Frame, Error = ConnSendError> + Unpin>(
205    mut writer: S,
206    client_secret_key: &SecretKey,
207    client_info: &ClientInfo,
208) -> Result<(), SendError> {
209    let msg = postcard::to_stdvec(client_info)?;
210    let signature = client_secret_key.sign(&msg);
211
212    writer
213        .send(Frame::ClientInfo {
214            client_public_key: client_secret_key.public(),
215            message: msg.into(),
216            signature,
217        })
218        .await?;
219    writer.flush().await?;
220    Ok(())
221}

223/// Reads the `FrameType::ClientInfo` frame from the client (its proof of identity)
224/// upon it's initial connection.
225#[cfg(any(test, feature = "server"))]
226pub(crate) async fn recv_client_key<E, S: Stream<Item = Result<Frame, E>> + Unpin>(
227    stream: S,
228) -> Result<(PublicKey, ClientInfo), E>
229where
230    E: From<RecvError>,
231{
232    // the client is untrusted at this point, limit the input size even smaller than our usual
233    // maximum frame size, and give a timeout
234
235    // TODO: variable recv size: 256 * 1024
236    let buf = tokio::time::timeout(
237        std::time::Duration::from_secs(10),
238        recv_frame(FrameType::ClientInfo, stream),
239    )
240    .await
241    .map_err(RecvError::from)??;
242
243    if let Frame::ClientInfo {
244        client_public_key,
245        message,
246        signature,
247    } = buf
248    {
249        client_public_key
250            .verify(&message, &signature)
251            .map_err(RecvError::from)?;
252
253        let info: ClientInfo = postcard::from_bytes(&message).map_err(RecvError::from)?;
254        Ok((client_public_key, info))
255    } else {
256        Err(UnexpectedFrameSnafu {
257            got: buf.typ(),
258            expected: FrameType::ClientInfo,
259        }
260        .build()
261        .into())
262    }
263}

265/// The relay protocol frames.
266#[derive(Debug, Clone, PartialEq, Eq)]
267pub(crate) enum Frame {
268    ClientInfo {
269        client_public_key: PublicKey,
270        message: Bytes,
271        signature: Signature,
272    },
273    SendPacket {
274        dst_key: PublicKey,
275        packet: Bytes,
276    },
277    RecvPacket {
278        src_key: PublicKey,
279        content: Bytes,
280    },
281    KeepAlive,
282    NotePreferred {
283        preferred: bool,
284    },
285    NodeGone {
286        node_id: PublicKey,
287    },
288    Ping {
289        data: [u8; 8],
290    },
291    Pong {
292        data: [u8; 8],
293    },
294    Health {
295        problem: Bytes,
296    },
297    Restarting {
298        reconnect_in: u32,
299        try_for: u32,
300    },
301}

303impl Frame {
304    pub(crate) fn typ(&self) -> FrameType {
305        match self {
306            Frame::ClientInfo { .. } => FrameType::ClientInfo,
307            Frame::SendPacket { .. } => FrameType::SendPacket,
308            Frame::RecvPacket { .. } => FrameType::RecvPacket,
309            Frame::KeepAlive => FrameType::KeepAlive,
310            Frame::NotePreferred { .. } => FrameType::NotePreferred,
311            Frame::NodeGone { .. } => FrameType::PeerGone,
312            Frame::Ping { .. } => FrameType::Ping,
313            Frame::Pong { .. } => FrameType::Pong,
314            Frame::Health { .. } => FrameType::Health,
315            Frame::Restarting { .. } => FrameType::Restarting,
316        }
317    }
318
319    /// Writes it self to the given buffer.
320    pub(crate) fn write_to<O: BufMut>(&self, mut dst: O) -> O {
321        dst.put_u8(self.typ().into());
322        match self {
323            Frame::ClientInfo {
324                client_public_key,
325                message,
326                signature,
327            } => {
328                dst.put(MAGIC.as_bytes());
329                dst.put(client_public_key.as_ref());
330                dst.put(&signature.to_bytes()[..]);
331                dst.put(&message[..]);
332            }
333            Frame::SendPacket { dst_key, packet } => {
334                dst.put(dst_key.as_ref());
335                dst.put(packet.as_ref());
336            }
337            Frame::RecvPacket { src_key, content } => {
338                dst.put(src_key.as_ref());
339                dst.put(content.as_ref());
340            }
341            Frame::KeepAlive => {}
342            Frame::NotePreferred { preferred } => {
343                if *preferred {
344                    dst.put_u8(PREFERRED);
345                } else {
346                    dst.put_u8(NOT_PREFERRED);
347                }
348            }
349            Frame::NodeGone { node_id: peer } => {
350                dst.put(peer.as_ref());
351            }
352            Frame::Ping { data } => {
353                dst.put(&data[..]);
354            }
355            Frame::Pong { data } => {
356                dst.put(&data[..]);
357            }
358            Frame::Health { problem } => {
359                dst.put(problem.as_ref());
360            }
361            Frame::Restarting {
362                reconnect_in,
363                try_for,
364            } => {
365                dst.put_u32(*reconnect_in);
366                dst.put_u32(*try_for);
367            }
368        }
369        dst
370    }

372    #[allow(clippy::result_large_err)]
373    pub(crate) fn from_bytes(bytes: Bytes, cache: &KeyCache) -> Result<Self, RecvError> {
374        if bytes.is_empty() {
375            return Err(TooSmallSnafu.build());
376        }
377        let frame_type = FrameType::from(bytes[0]);
378        let content = bytes.slice(1..);
379        let res = match frame_type {
380            FrameType::ClientInfo => {
381                if content.len() < PublicKey::LENGTH + Signature::BYTE_SIZE + MAGIC.len() {
382                    return Err(InvalidFrameSnafu.build());
383                }
384                if &content[..MAGIC.len()] != MAGIC.as_bytes() {
385                    return Err(InvalidFrameSnafu.build());
386                }
387
388                let start = MAGIC.len();
389                let client_public_key =
390                    cache.key_from_slice(&content[start..start + PublicKey::LENGTH])?;
391                let start = start + PublicKey::LENGTH;
392                let signature =
393                    Signature::from_slice(&content[start..start + Signature::BYTE_SIZE])?;
394                let start = start + Signature::BYTE_SIZE;
395                let message = content.slice(start..);
396                Self::ClientInfo {
397                    client_public_key,
398                    message,
399                    signature,
400                }
401            }
402            FrameType::SendPacket => {
403                if content.len() < PublicKey::LENGTH {
404                    return Err(InvalidFrameSnafu.build());
405                }
406                let frame_len = content.len() - PublicKey::LENGTH;
407                if frame_len > MAX_PACKET_SIZE {
408                    return Err(FrameTooLargeSnafu { frame_len }.build());
409                }
410
411                let dst_key = cache.key_from_slice(&content[..PublicKey::LENGTH])?;
412                let packet = content.slice(PublicKey::LENGTH..);
413                Self::SendPacket { dst_key, packet }
414            }
415            FrameType::RecvPacket => {
416                if content.len() < PublicKey::LENGTH {
417                    return Err(InvalidFrameSnafu.build());
418                }
419
420                let frame_len = content.len() - PublicKey::LENGTH;
421                if frame_len > MAX_PACKET_SIZE {
422                    return Err(FrameTooLargeSnafu { frame_len }.build());
423                }
424
425                let src_key = cache.key_from_slice(&content[..PublicKey::LENGTH])?;
426                let content = content.slice(PublicKey::LENGTH..);
427                Self::RecvPacket { src_key, content }
428            }
429            FrameType::KeepAlive => {
430                if !content.is_empty() {
431                    return Err(InvalidFrameSnafu.build());
432                }
433                Self::KeepAlive
434            }
435            FrameType::NotePreferred => {
436                if content.len() != 1 {
437                    return Err(InvalidFrameSnafu.build());
438                }
439                let preferred = match content[0] {
440                    PREFERRED => true,
441                    NOT_PREFERRED => false,
442                    _ => return Err(InvalidFrameSnafu.build()),
443                };
444                Self::NotePreferred { preferred }
445            }
446            FrameType::PeerGone => {
447                if content.len() != PublicKey::LENGTH {
448                    return Err(InvalidFrameSnafu.build());
449                }
450                let peer = cache.key_from_slice(&content[..32])?;
451                Self::NodeGone { node_id: peer }
452            }
453            FrameType::Ping => {
454                if content.len() != 8 {
455                    return Err(InvalidFrameSnafu.build());
456                }
457                let mut data = [0u8; 8];
458                data.copy_from_slice(&content[..8]);
459                Self::Ping { data }
460            }
461            FrameType::Pong => {
462                if content.len() != 8 {
463                    return Err(InvalidFrameSnafu.build());
464                }
465                let mut data = [0u8; 8];
466                data.copy_from_slice(&content[..8]);
467                Self::Pong { data }
468            }
469            FrameType::Health => Self::Health { problem: content },
470            FrameType::Restarting => {
471                if content.len() != 4 + 4 {
472                    return Err(InvalidFrameSnafu.build());
473                }
474                let reconnect_in = u32::from_be_bytes(
475                    content[..4]
476                        .try_into()
477                        .map_err(|_| InvalidFrameSnafu.build())?,
478                );
479                let try_for = u32::from_be_bytes(
480                    content[4..]
481                        .try_into()
482                        .map_err(|_| InvalidFrameSnafu.build())?,
483                );
484                Self::Restarting {
485                    reconnect_in,
486                    try_for,
487                }
488            }
489            _ => {
490                return Err(InvalidFrameTypeSnafu { frame_type }.build());
491            }
492        };
493        Ok(res)
494    }
495}

497/// Receives the next frame and matches the frame type. If the correct type is found returns the content,
498/// otherwise an error.
499#[cfg(any(test, feature = "server"))]
500pub(crate) async fn recv_frame<E, S: Stream<Item = Result<Frame, E>> + Unpin>(
501    frame_type: FrameType,
502    mut stream: S,
503) -> Result<Frame, E>
504where
505    RecvError: Into<E>,
506{
507    match stream.next().await {
508        Some(Ok(frame)) => {
509            if frame_type != frame.typ() {
510                return Err(UnexpectedFrameSnafu {
511                    got: frame.typ(),
512                    expected: frame_type,
513                }
514                .build()
515                .into());
516            }
517            Ok(frame)
518        }
519        Some(Err(err)) => Err(err),
520        None => Err(RecvError::from(std::io::Error::new(
521            std::io::ErrorKind::UnexpectedEof,
522            "expected frame".to_string(),
523        ))
524        .into()),
525    }
526}
