19/// A Stream and Sink for [`Frame`]s connected to a single relay client.
20///
21/// The stream receives message from the client while the sink sends them to the client.
22#[derive(Debug)]
23pub(crate) enum RelayedStream {
24    Relay(Framed<MaybeTlsStream, RelayCodec>),
25    Ws(WebSocketStream<MaybeTlsStream>, KeyCache),
26}
27
28fn ws_to_io_err(e: tokio_websockets::Error) -> std::io::Error {
29    match e {
30        tokio_websockets::Error::Io(io_err) => io_err,
31        _ => std::io::Error::other(e.to_string()),
32    }
33}
34
35impl Sink<Frame> for RelayedStream {
36    type Error = std::io::Error;
37
38    fn poll_ready(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
39        match *self {
40            Self::Relay(ref mut framed) => Pin::new(framed).poll_ready(cx),
41            Self::Ws(ref mut ws, _) => Pin::new(ws).poll_ready(cx).map_err(ws_to_io_err),
42        }
43    }
44
45    fn start_send(mut self: Pin<&mut Self>, item: Frame) -> Result<(), Self::Error> {
46        match *self {
47            Self::Relay(ref mut framed) => Pin::new(framed).start_send(item),
48            Self::Ws(ref mut ws, _) => Pin::new(ws)
49                .start_send(tokio_websockets::Message::binary(
50                    tokio_websockets::Payload::from(item.encode_for_ws_msg()),
51                ))
52                .map_err(ws_to_io_err),
53        }
54    }
55
56    fn poll_flush(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
57        match *self {
58            Self::Relay(ref mut framed) => Pin::new(framed).poll_flush(cx),
59            Self::Ws(ref mut ws, _) => Pin::new(ws).poll_flush(cx).map_err(ws_to_io_err),
60        }
61    }
62
63    fn poll_close(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
64        match *self {
65            Self::Relay(ref mut framed) => Pin::new(framed).poll_close(cx),
66            Self::Ws(ref mut ws, _) => Pin::new(ws).poll_close(cx).map_err(ws_to_io_err),
67        }
68    }
69}

71/// Relay stream errors
72#[derive(Debug, Snafu)]
73#[non_exhaustive]
74pub enum StreamError {
75    #[snafu(transparent)]
76    Proto { source: RecvError },
77    #[snafu(transparent)]
78    Ws { source: tokio_websockets::Error },
79}
80
81impl Stream for RelayedStream {
82    type Item = Result<Frame, StreamError>;
83
84    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
85        match *self {
86            Self::Relay(ref mut framed) => Pin::new(framed).poll_next(cx).map_err(Into::into),
87            Self::Ws(ref mut ws, ref cache) => match Pin::new(ws).poll_next(cx) {
88                Poll::Ready(Some(Ok(msg))) => {
89                    if msg.is_close() {
90                        // Indicate the stream is done when we receive a close message.
91                        // Note: We don't have to poll the stream to completion for it to close gracefully.
92                        return Poll::Ready(None);
93                    }
94                    if !msg.is_binary() {
95                        tracing::warn!(
96                            ?msg,
97                            "Got websocket message of unsupported type, skipping."
98                        );
99                        return Poll::Pending;
100                    }
101                    let frame = Frame::decode_from_ws_msg(msg.into_payload().into(), cache)
102                        .map_err(Into::into);
103                    Poll::Ready(Some(frame))
104                }
105                Poll::Ready(Some(Err(e))) => Poll::Ready(Some(Err(e.into()))),
106                Poll::Ready(None) => Poll::Ready(None),
107                Poll::Pending => Poll::Pending,
108            },
109        }
110    }
111}

113/// The main underlying IO stream type used for the relay server.
114///
115/// Allows choosing whether or not the underlying [`tokio::net::TcpStream`] is served over Tls
116#[derive(Debug)]
117#[allow(clippy::large_enum_variant)]
118pub enum MaybeTlsStream {
119    /// A plain non-Tls [`tokio::net::TcpStream`]
120    Plain(tokio::net::TcpStream),
121    /// A Tls wrapped [`tokio::net::TcpStream`]
122    Tls(tokio_rustls::server::TlsStream<tokio::net::TcpStream>),
123    /// An in-memory bidirectional pipe.
124    #[cfg(test)]
125    Test(tokio::io::DuplexStream),
126}
127
128impl AsyncRead for MaybeTlsStream {
129    fn poll_read(
130        mut self: Pin<&mut Self>,
131        cx: &mut Context<'_>,
132        buf: &mut tokio::io::ReadBuf<'_>,
133    ) -> Poll<std::io::Result<()>> {
134        match &mut *self {
135            MaybeTlsStream::Plain(ref mut s) => Pin::new(s).poll_read(cx, buf),
136            MaybeTlsStream::Tls(ref mut s) => Pin::new(s).poll_read(cx, buf),
137            #[cfg(test)]
138            MaybeTlsStream::Test(ref mut s) => Pin::new(s).poll_read(cx, buf),
139        }
140    }
141}
